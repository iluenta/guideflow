import { TinyColor } from '@ctrl/tinycolor';

export function harmonizeThemeFromPrimary(
  baseTheme: Theme,
  newPrimaryColor: string
): Theme {
  const primary = new TinyColor(newPrimaryColor);
  
  // Validar que el color tenga suficiente contraste
  if (primary.getLuminance() > 0.7) {
    throw new Error('El color es demasiado claro. Elige un tono m√°s oscuro.');
  }
  
  // Generar color secundario (complementario o an√°logo)
  const secondary = primary
    .spin(30) // Rotar 30¬∞ en la rueda de color
    .saturate(10)
    .lighten(10);
  
  // Generar color de acento (tri√°dico)
  const accent = primary
    .spin(120)
    .saturate(20);
  
  // Generar escala de grises neutral basada en el primario
  const neutral = generateNeutralScale(primary);
  
  // Calcular colores de texto con contraste WCAG AA
  const textPrimary = primary.isLight() 
    ? '#1a1a1a' 
    : primary.darken(40).toString();
  
  const textSecondary = primary.isLight()
    ? '#4a4a4a'
    : primary.darken(30).toString();
  
  return {
    ...baseTheme,
    colors: {
      primary: primary.toHexString(),
      secondary: secondary.toHexString(),
      accent: accent.toHexString(),
      neutral: neutral,
      background: primary.getLuminance() > 0.5 ? '#fefefe' : '#ffffff',
      surface: '#ffffff',
      text: {
        primary: textPrimary,
        secondary: textSecondary,
        muted: neutral[400]
      }
    }
  };
}

function generateNeutralScale(baseColor: TinyColor) {
  const hue = baseColor.toHsl().h;
  const saturation = 5; // Muy desaturado para neutrales
  
  return {
    50: new TinyColor({ h: hue, s: saturation, l: 98 }).toHexString(),
    100: new TinyColor({ h: hue, s: saturation, l: 96 }).toHexString(),
    200: new TinyColor({ h: hue, s: saturation, l: 92 }).toHexString(),
    300: new TinyColor({ h: hue, s: saturation, l: 84 }).toHexString(),
    400: new TinyColor({ h: hue, s: saturation, l: 68 }).toHexString(),
    500: new TinyColor({ h: hue, s: saturation, l: 52 }).toHexString(),
    600: new TinyColor({ h: hue, s: saturation, l: 40 }).toHexString(),
    700: new TinyColor({ h: hue, s: saturation, l: 30 }).toHexString(),
    800: new TinyColor({ h: hue, s: saturation, l: 20 }).toHexString(),
    900: new TinyColor({ h: hue, s: saturation, l: 12 }).toHexString()
  };
}

// Validar contraste WCAG
export function validateContrast(
  foreground: string, 
  background: string
): { passes: boolean; ratio: number } {
  const fg = new TinyColor(foreground);
  const bg = new TinyColor(background);
  
  const ratio = TinyColor.readability(fg, bg);
  const passes = ratio >= 4.5; // WCAG AA para texto normal
  
  return { passes, ratio };
}
